SELECT u.name, COUNT(o._id) as order_count
FROM users u
LEFT JOIN orders o ON u._id = o.userId
WHERE u.status = 'active'
  AND o.items[0].price > 100
GROUP BY u.name
ORDER BY order_count DESC
LIMIT 5
